#!/bin/bash

# =====================================================
# Лабораторная работа. Вариант 1
# Скрипт: backup_images.sh
# Назначение: копирование всех изображений в папку резервного копирования
# Автор: студент
# =====================================================

# Переменная SOURCE — откуда копируем
# $1 — первый параметр скрипта
# Если параметр не передан (пусто), используем текущий каталог "."
SOURCE="${1:-.}"

# Переменная TARGET — куда копируем
# $2 — второй параметр скрипта
# Если не передан, используем ~/backup (папка backup в домашнем каталоге)
TARGET="${2:-$HOME/backup}"

# Расширения файлов, которые считаем изображениями
# iname ищет без учёта регистра (JPG = jpg)
EXTENSIONS="jpg jpeg png gif bmp tiff webp svg"

# Создаём целевую папку, если её нет
# -p создаёт и родительские каталоги при необходимости
mkdir -p "$TARGET"

# Выводим информацию о начале работы
echo "================================"
echo "Резервное копирование изображений"
echo "Источник: $SOURCE"
echo "Цель: $TARGET"
echo "================================"

# Счётчик скопированных файлов
count=0

# Цикл по всем расширениям из списка
for ext in $EXTENSIONS; do
    
    # find — поиск файлов
    # "$SOURCE" — в каком каталоге ищем
    # -type f — ищем только файлы (не каталоги)
    # -iname "*.$ext" — имя заканчивается на .ext (регистронезависимо)
    # while read — читаем каждую найденную строку
    # cp — копируем файл в целевую папку
    # -v — подробный вывод (verbose)
    
    find "$SOURCE" -type f -iname "*.$ext" | while read -r file; do
        cp -v "$file" "$TARGET/"
        
        # Увеличиваем счётчик (но из-за пайпа счётчик не сохраняется,
        # это демонстрационный вариант. В реальности нужно использовать
        # другой подход для подсчёта)
        ((count++))
    done
done

echo "copy image: $count"
